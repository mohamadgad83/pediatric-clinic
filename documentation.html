🗄️ هيكل قاعدة البيانات

جدول clinic_profiles

CREATE TABLE clinic_profiles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    full_name TEXT NOT NULL,
    role TEXT NOT NULL CHECK (role IN ('doctor', 'assistant')),
    phone TEXT,
    avatar_url TEXT,
    username TEXT UNIQUE NOT NULL,
    email TEXT UNIQUE,
    password_hash TEXT,
    confirmed BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now()
);

جدول clinic_patients

CREATE TABLE clinic_patients (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL,
    birth_date DATE NOT NULL,
    gender TEXT CHECK (gender IN ('male', 'female')),
    parent_name TEXT NOT NULL,
    parent_phone TEXT NOT NULL,
    parent_email TEXT,
    address TEXT,
    custom_fields JSONB DEFAULT '{}'::jsonb,
    created_by UUID REFERENCES clinic_profiles(id),
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now()
);

جدول clinic_appointments

CREATE TABLE clinic_appointments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id UUID NOT NULL REFERENCES clinic_patients(id) ON DELETE CASCADE,
    appointment_date DATE NOT NULL,
    appointment_time TIME,
    queue_number INTEGER NOT NULL,
    status TEXT DEFAULT 'waiting' CHECK (status IN ('waiting', 'in_clinic', 'done', 'cancelled')),
    vital_signs JSONB DEFAULT '{}'::jsonb,
    chief_complaint TEXT,
    notes TEXT,
    created_by UUID REFERENCES clinic_profiles(id),
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now()
);

جدول clinic_medical_records

CREATE TABLE clinic_medical_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id UUID NOT NULL REFERENCES clinic_patients(id) ON DELETE CASCADE,
    appointment_id UUID REFERENCES clinic_appointments(id) ON DELETE SET NULL,
    diagnosis TEXT NOT NULL,
    prescription JSONB DEFAULT '[]'::jsonb,
    notes TEXT,
    attachments TEXT[],
    created_by UUID REFERENCES clinic_profiles(id),
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now()
);

دالة التحقق من كلمة المرور

CREATE OR REPLACE FUNCTION verify_user(
    p_identifier TEXT,
    p_password TEXT
)
RETURNS TABLE(
    user_id UUID,
    full_name TEXT,
    user_role TEXT,
    user_email TEXT,
    user_username TEXT
) AS $$
DECLARE user_record RECORD;
BEGIN
    SELECT id, full_name, role, email, username, password_hash
    INTO user_record
    FROM clinic_profiles
    WHERE username = p_identifier OR email = p_identifier;

    IF user_record.id IS NULL THEN RETURN; END IF;

    IF user_record.password_hash = crypt(p_password, user_record.password_hash) THEN
        user_id := user_record.id;
        full_name := user_record.full_name;
        user_role := user_record.role;
        user_email := user_record.email;
        user_username := user_record.username;
        RETURN NEXT;
    END IF;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

🔐 نظام المصادقة

  • API: app/api/login/route.ts – يستقبل identifier (username/email) و password، ويتحقق من المستخدم في clinic_profiles، ويعيد بيانات المستخدم مع Cookies.
  • Middleware: middleware.ts – يتحقق من Cookies ويحمي المسارات (/doctor, /assistant) ويدير الصلاحيات.
  • Hook: hooks/useAuth.ts – يقرأ بيانات المستخدم من localStorage ويوفر دالة تسجيل الخروج.
